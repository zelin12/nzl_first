#include<iostream>
#include<vector>
#include<algorithm>

using namespace std;

class Solution {
public:
	vector<vector<int>> threeSum(vector<int>& nums)
	{
		sort(nums.begin(), nums.begin() + nums.size());
		int first_p, second_p, third_p;
		vector<vector<int>> array2 = {};
		vector<int> array3;
		if (nums.size() < 3)
		{
			array2 = {};
			return array2;
		}
		for (first_p = 0; first_p < nums.size() - 2; first_p++)
		{
			int second_p = first_p + 1;
			int third_p = nums.size() - 1;
			while (second_p != third_p) 
			{
				if ((nums[second_p] + nums[third_p] + nums[first_p]) == 0)
				{
					array3.push_back(nums[first_p]);
					array3.push_back(nums[second_p]);
					array3.push_back(nums[third_p]);
					array2.push_back(array3);
					array3.clear();
					third_p--;
					second_p++;
					break;
				}
				if ((nums[second_p] + nums[third_p] + nums[first_p]) > 0)
				{
					third_p--;
					continue;
				}
				if ((nums[second_p] + nums[third_p] + nums[first_p]) < 0)
				{
					second_p++;
					continue;
				}
			} 
		}
		return array2;
	}
};

	vector<vector<int>> b = a.threeSum(array1);
