#include<iostream>
#include<vector>
#include<algorithm>

using namespace std;


class Solution {
public:
	int maxArea(vector<int>& height) 
	{
		int maxNumber = 0;
		int j = height.size() - 1;
		for (int i = 0; i != j;)
		{
			if (height[i] <= height[j])
			{
				maxNumber = max(maxNumber, height[i] * (j - i));
				i++;
			}
			else 
			{
				maxNumber = max(maxNumber, height[j] * (j - i));
				j--;
			}
		}
		return maxNumber;
	}
};
int main()
{
	vector<int> array1 = { 1, 8, 6, 2, 5, 4, 8, 3, 7 };
	Solution a;
	a.maxArea(array1);
	
	return 0;
}
