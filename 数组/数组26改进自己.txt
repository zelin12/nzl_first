class Solution {
public:
	int removeDuplicates(vector<int>& nums) {
		int number = nums[0];
		int count = 1;
		int index = 0;
		for (int i = 0; i < nums.size(); i++)
		{
			if (number == nums[i])
			{
				continue;
			}
			else
			{
				nums[++index] = nums[i];
				count++;
				number = nums[i];
			}
		}
		return count;
	}
};
