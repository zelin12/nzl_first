//注意循环及时跳出，还是写得少
//注意初始化时最好采用不在此定义或者应用中的数值
bool mycomp(int i, int j) {
    return (i< j);
}
class Solution {
public:
    int missingNumber(vector<int>& nums) {
        int a = -1;
        sort(nums.begin(),nums.end(),mycomp);
        for(int i = 0;i < nums.size();i++)
        {
            if(i != nums[i])
            {
                a = i;
                break;
            }
        }
        if(a == -1 )
          a = nums.size();
        return a;
    }
};
