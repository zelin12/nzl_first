/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
//暴力解法，直接遍历看是否相同
//这里也要注意最底下的节点下也是null，写null返回值时需要注意
  void search(TreeNode* x, vector<int>& at)
  {
	  if (x == nullptr)
	  {
          at.push_back(0);
		  return;
	  }
      at.push_back(x->val);
	  search(x->left, at);
	  search(x->right, at);
  }
  class Solution {
  public:
	  bool isSameTree(TreeNode* p, TreeNode* q) 
	  {
		  vector<int> a, b;
		  search(p, a);
		  search(q, b);
		  if (a == b)
			  return true;
		  else
			  return false;
	  }
  };
