#include<iostream>

using namespace std;

//辅助函数
void print_arr(int arr[], int n)
{
	for (int i = 0; i < n; i++)
	{
		cout << arr[i] << endl;
	}
}

void merge(int arr[], int tempArr[], int left,int mid, int right)
{
	//标记左半区第一个未被排序的元素
	int l_pos = left;
	//标记右半区第一个未被排序的元素
	int r_pos = mid + 1;
	//临时数组的下标
	int pos = left;

	//合并
	while (l_pos <= mid && r_pos <= right)
	{
		if (arr[l_pos] < arr[r_pos])
			tempArr[pos++] = arr[l_pos++];
		else
			tempArr[pos++] = arr[r_pos++];
	}

	//合并左半区剩余元素
	while (l_pos <= mid)
		tempArr[pos++] = arr[l_pos++];
	//合并右半区剩余元素
	while (r_pos <= right)
		tempArr[pos++] = arr[r_pos++];

	//把合并后的数组拷贝回原数组
	while (left <= right)
	{
		arr[left] = tempArr[left];
		left++;
	}
}


//归并排序
void msort(int arr[], int tempArr[], int left, int right)
{
	//如果只有一个元素则不需要排序
	//只有一个元素的区域，本来就是有序的，只需要被归并即可
	if (left < right)
	{
		//找中间点
		int mid = (left + right) / 2;
		//递归划分左半区
		msort(arr, tempArr, left, mid);
		//递归划分右半区
		msort(arr, tempArr, mid + 1,right);
		//合并已经排序的部分
		merge(arr, tempArr, left, mid, right);
	}
	
}

void merge_sort(int arr[], int n)
{
	//分配一个辅助数组
	int* p = new int(n);
	if (p)//辅助数组分配成功
	{
		msort(arr, p, 0, n - 1);
//		delete p;
	}
	else {
		cout << "error" << endl;
	}

}

int main()
{
	int arr[] = { 9,5,2,7,12,4,3,1,11 };
	int n = 9;
	print_arr(arr, n);

	merge_sort(arr, n);
	print_arr(arr, n);
	return 0;
}
